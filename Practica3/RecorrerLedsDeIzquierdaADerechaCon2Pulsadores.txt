MAIN_PROG CODE                     

START
    BANKSEL OSCCON
    MOVLW 0x71
    MOVWF OSCCON
    MOVLW 0x00
    MOVWF TRISA
    BSF STATUS,0
    BSF TRISB,5
    BSF TRISB,6
    BANKSEL ANSEL
    MOVLW 0x00
    MOVWF ANSEL
    BANKSEL PORTA
    CLRF PORTA
INICIO
    
    BTFSC PORTB, 5
    GOTO SUMA
    BTFSC PORTB, 6
    GOTO RESTA
    GOTO INICIO
RESTA
    RRF TRISA, 1
    GOTO INICIO    
SUMA
    RLF TRISA, 1
    GOTO INICIO
    END
